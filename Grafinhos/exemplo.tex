  Exemplo de prova
\begin{align*}
  2 * |A(G)| &= \sum_{w \in V(G)} g(w)\\ &< n + \sum_{w \in V(G) \setminus \{u, v\}} g(w)\\
                                      &= 2*\Big(\frac{(n - 1)(n - 2)}{2} + 2\Big)\\
                                      &= 2*|A(G)|
\end{align*}
\end{proof}
